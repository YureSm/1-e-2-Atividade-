  <p id="resultado"></p>

  <script>
    function calcularDesconto() {
      const valor = parseFloat(document.getElementById('valor').value);
      const resultado = document.getElementById('resultado');
      
      if (isNaN(valor) || valor <= 0) {
        resultado.textContent = 'Informe um valor vÃ¡lido.';
        return;
      }

      let desconto = 0;
      if (valor > 200) {
        desconto = valor * 0.10;
      } else {
        desconto = valor * 0.05;
      }

      const final = valor - desconto;
      resultado.textContent = `Valor com desconto: R$ ${final.toFixed(2)}`;
    }
  </script>
