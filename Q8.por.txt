        <label for="largura">Largura:</label>
        <input type="number" id="largura" placeholder="Digite a largura">
        <br>
        <label for="altura">Altura:</label>
        <input type="number" id="altura" placeholder="Digite a altura">
        <br>
        <button onclick="calcularVolume()">Calcular Volume</button>
        <div class="result" id="resultado"></div>
    </div>

    <script>
        function calcularVolume() {
            var c = parseFloat(document.getElementById('comprimento').value);
            var l = parseFloat(document.getElementById('largura').value);
            var a = parseFloat(document.getElementById('altura').value);
            var volume = c * l * a;
            document.getElementById('resultado').textContent = "Volume: " + volume;
        }
    </script>
