<body>
  <h2>Verificar se o número está entre 10 e 50</h2>
  <label>Digite um número: <input type="number" id="numero"></label><br>
  <button onclick="verificar()">Verificar</button>
  <p id="resultado"></p>

  <script>
    function verificar() {
      const numero = parseInt(document.getElementById('numero').value);
      const resultado = document.getElementById('resultado');

      if (numero >= 10 && numero <= 50) {
        resultado.textContent = 'Está entre 10 e 50';
      } else {
        resultado.textContent = 'Fora do intervalo';
      }
    }
  </script>
