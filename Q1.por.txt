<body>
    <h2>Soma de Dois Números</h2>
    <label>Digite o primeiro número: <input type="number" id="num1"></label><br><br>
    <label>Digite o segundo número: <input type="number" id="num2"></label><br><br>
    <button onclick="calcularSoma()">Somar</button>
    <p id="resultado"></p>

    <script>
        function calcularSoma() {
            const a = parseInt(document.getElementById("num1").value);
            const b = parseInt(document.getElementById("num2").value);
            const soma = a + b;
            document.getElementById("resultado").innerText = "A soma é: " + soma;
        }
    </script>
