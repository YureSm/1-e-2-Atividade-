<body>
    <h2>Digite os elementos da matriz 3x3:</h2>
    <div class="matriz" id="inputs">
        <!-- Inputs da matriz -->
        <script>
            for (let i = 0; i < 3; i++) {
                for (let j = 0; j < 3; j++) {
                    document.write(`<input type="number" id="m${i}${j}" placeholder="[${i}][${j}]">`);
                }
                document.write("<br>");
            }
        </script>
    </div>
    <button onclick="mostrarMatriz()">Exibir Matriz</button>
    <h3>Matriz digitada:</h3>
    <div id="resultado"></div>

    <script>
        function mostrarMatriz() {
            let matriz = [];
            let output = "";

            for (let i = 0; i < 3; i++) {
                matriz[i] = [];
                for (let j = 0; j < 3; j++) {
                    const valor = document.getElementById(`m${i}${j}`).value;
                    matriz[i][j] = valor ? parseInt(valor) : 0;
                    output += matriz[i][j] + " ";
                }
                output += "<br>";
            }

            document.getElementById("resultado").innerHTML = output;
        }
    </script>
