<body>
    <h1>Tabuada</h1>
    <label for="numero">Digite um número:</label>
    <input type="number" id="numero">
    <button onclick="mostrarTabuada()">Mostrar Tabuada</button>

    <div class="resultado" id="resultado"></div>

    <script>
        function mostrarTabuada() {
            const num = parseInt(document.getElementById('numero').value);
            const resultadoDiv = document.getElementById('resultado');
            resultadoDiv.innerHTML = ''; // Limpa resultado anterior

            if (isNaN(num)) {
                resultadoDiv.innerHTML = 'Por favor, digite um número válido.';
                return;
            }

            for (let i = 1; i <= 10; i++) {
                const linha = document.createElement('p');
                linha.textContent = `${num} x ${i} = ${num * i}`;
                resultadoDiv.appendChild(linha);
            }
        }
    </script>
