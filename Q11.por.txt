    <input type="number" id="numero" />
    <button onclick="verificarParOuImpar()">Verificar</button>
    <p id="resultado"></p>

    <script>
        function verificarParOuImpar() {
            const num = parseInt(document.getElementById('numero').value);
            const resultado = document.getElementById('resultado');
            
            if (isNaN(num)) {
                resultado.textContent = "Por favor, digite um número válido.";
            } else if (num % 2 === 0) {
                resultado.textContent = "Número par";
            } else {
                resultado.textContent = "Número ímpar";
            }
        }
    </script>
