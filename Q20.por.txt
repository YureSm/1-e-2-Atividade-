    <meta charset="UTF-8">
    <title>Verificar Múltiplos de 3 e 5</title>
</head>
<body>
    <h2>Digite um número:</h2>
    <input type="number" id="numero">
    <button onclick="verificarMultiplo()">Verificar</button>
    <p id="resultado"></p>

    <script>
        function verificarMultiplo() {
            const numero = parseInt(document.getElementById("numero").value);
            const resultado = document.getElementById("resultado");

            if (numero % 3 === 0 && numero % 5 === 0) {
                resultado.textContent = "É múltiplo de 3 e 5";
            } else {
                resultado.textContent = "Não é múltiplo de 3 e 5";
            }
        }
    </script>
