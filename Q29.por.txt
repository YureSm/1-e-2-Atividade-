        <button onclick="calcularFatorial()">Calcular Fatorial</button>
        <p id="resultado"></p>
    </div>

    <script>
        function calcularFatorial() {
            var num = parseInt(document.getElementById("num").value);
            var fat = 1;
            
            if (num < 0) {
                document.getElementById("resultado").innerText = "Por favor, insira um número não negativo.";
                return;
            }

            for (var i = 1; i <= num; i++) {
                fat *= i;
            }

            document.getElementById("resultado").innerText = "Fatorial: " + fat;
        }
    </script>
