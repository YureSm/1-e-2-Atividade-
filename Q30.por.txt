  <h2>Verificador de Número Primo</h2>
  <label for="numero">Digite um número:</label>
  <input type="number" id="numero">
  <button onclick="verificarPrimo()">Verificar</button>
  <p id="resultado"></p>

  <script>
    function verificarPrimo() {
      const num = parseInt(document.getElementById('numero').value);
      let cont = 0;

      for (let i = 1; i <= num; i++) {
        if (num % i === 0) {
          cont++;
        }
      }

      const resultado = document.getElementById('resultado');
      if (cont === 2) {
        resultado.textContent = "Número primo";
      } else {
        resultado.textContent = "Não é primo";
      }
    }
  </script>
