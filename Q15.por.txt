  <h2>Comparar dois valores</h2>
  <label>Valor A: <input type="number" id="valorA"></label><br>
  <label>Valor B: <input type="number" id="valorB"></label><br>
  <button onclick="comparar()">Comparar</button>
  <p id="resultado"></p>

  <script>
    function comparar() {
      const a = parseFloat(document.getElementById('valorA').value);
      const b = parseFloat(document.getElementById('valorB').value);
      const resultado = document.getElementById('resultado');

      if (a > b) {
        resultado.textContent = 'Maior: ' + a;
      } else if (b > a) {
        resultado.textContent = 'Maior: ' + b;
      } else {
        resultado.textContent = 'Iguais';
      }
    }
  </script>
