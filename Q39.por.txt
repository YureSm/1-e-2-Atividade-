    <script>
        function calcularQuadrados() {
            // Matriz de entrada
            var matriz = [
                [document.getElementById('matriz00').value, document.getElementById('matriz01').value, document.getElementById('matriz02').value],
                [document.getElementById('matriz10').value, document.getElementById('matriz11').value, document.getElementById('matriz12').value],
                [document.getElementById('matriz20').value, document.getElementById('matriz21').value, document.getElementById('matriz22').value]
            ];

            // Matriz dos quadrados
            var quadrado = [
                [Math.pow(matriz[0][0], 2), Math.pow(matriz[0][1], 2), Math.pow(matriz[0][2], 2)],
                [Math.pow(matriz[1][0], 2), Math.pow(matriz[1][1], 2), Math.pow(matriz[1][2], 2)],
                [Math.pow(matriz[2][0], 2), Math.pow(matriz[2][1], 2), Math.pow(matriz[2][2], 2)]
            ];

            // Exibindo os resultados
            for (var i = 0; i < 3; i++) {
                for (var j = 0; j < 3; j++) {
                    document.getElementById('quadrado' + i + j).textContent = quadrado[i][j];
                }
            }
        }
    </script>
