    <label for="nota2">Nota 2: </label>
    <input type="number" id="nota2" required><br><br>

    <label for="nota3">Nota 3: </label>
    <input type="number" id="nota3" required><br><br>

    <button onclick="calcularMedia()">Calcular Média</button>

    <h2 id="resultado"></h2>

    <script>
        function calcularMedia() {
            var n1 = parseFloat(document.getElementById("nota1").value);
            var n2 = parseFloat(document.getElementById("nota2").value);
            var n3 = parseFloat(document.getElementById("nota3").value);
            
            if (isNaN(n1) || isNaN(n2) || isNaN(n3)) {
                alert("Por favor, insira todas as notas corretamente.");
                return;
            }

            var media = (n1 + n2 + n3) / 3;
            document.getElementById("resultado").innerText = "Média: " + media.toFixed(2);
        }
    </script>
