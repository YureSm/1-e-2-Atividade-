    <h1>Calculadora de Soma</h1>
    <p>Digite números para somar. Digite 0 para sair.</p>
    
    <label for="num">Digite um número:</label>
    <input type="number" id="num" name="num">
    <button onclick="adicionarNumero()">Adicionar</button>

    <h2>Soma Total: <span id="soma">0</span></h2>

    <script>
        let soma = 0;

        function adicionarNumero() {
            let num = parseInt(document.getElementById("num").value);
            if (num === 0) {
                alert("Soma total: " + soma);
                soma = 0; // Resetando a soma
                document.getElementById("soma").textContent = soma;
            } else {
                soma += num;
                document.getElementById("soma").textContent = soma;
            }
            document.getElementById("num").value = ""; // Limpar o campo de entrada
        }
    </script>
